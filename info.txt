## iniciar projeto
npm init -y 

##configurar npm - fixar versão das dependencias
.npmrc => save-exact=true

## instalar dependencias dev
# tsx - rodar códico ts
# tsup - preparar build
npm i typescript @types/node tsx tsup -D

## configuração ts
npx tsc --init

# fastify - server http
npm i fastify

# scripts 
"start:dev":"tsx watch src/server.ts"
"start":"node build/server.js",
"build":"tsup src --out-dir build"

# lib - dotenv
npm i dotenv

# lib - zod para validações
npm i zod

# validação de envs com zod.

## EsLint
npm i -D eslint 
npx eslint --init

npm i -D @rocketseat/eslint-config 

### ORM
#instalar prisma
npm -i prisma -D
#inicia configurações
npx prisma init
#client
npm i @prisma/client

#subir banco e executar migration
npx prisma migrate dev

npx prisma migrate deploy -- ambiente prods

npx prisma studio
